## 2. CSV DATASET GENERATION EXAMPLES

### Example 4: Generate Sample Sales Data CSV
curl -X POST http://localhost:8000/v1/execute \
  -H "Content-Type: application/json" \
  -d '{
    "code": "import csv\nimport random\nfrom datetime import datetime, timedelta\n\n# Generate sample sales data\ndata = []\nproducts = [\"Laptop\", \"Phone\", \"Tablet\", \"Headphones\", \"Mouse\"]\nregions = [\"North\", \"South\", \"East\", \"West\"]\n\nfor i in range(100):\n    date = datetime.now() - timedelta(days=random.randint(0, 365))\n    row = {\n        \"date\": date.strftime(\"%Y-%m-%d\"),\n        \"product\": random.choice(products),\n        \"region\": random.choice(regions),\n        \"sales\": random.randint(100, 10000),\n        \"quantity\": random.randint(1, 50)\n    }\n    data.append(row)\n\n# Write to CSV\nwith open(\"sales_data.csv\", \"w\", newline=\"\") as f:\n    writer = csv.DictWriter(f, fieldnames=[\"date\", \"product\", \"region\", \"sales\", \"quantity\"])\n    writer.writeheader()\n    writer.writerows(data)\n\nprint(f\"Generated {len(data)} sales records in sales_data.csv\")",
    "language": "python"
  }'

### Example 5: Generate Weather Data Dataset
curl -X POST http://localhost:8000/v1/execute \
  -H "Content-Type: application/json" \
  -d '{
    "code": "import csv\nimport random\nfrom datetime import datetime, timedelta\n\n# Generate weather data\ncities = [\"New York\", \"London\", \"Tokyo\", \"Sydney\", \"Paris\"]\ndata = []\n\nstart_date = datetime(2023, 1, 1)\nfor i in range(365):\n    for city in cities:\n        temp = random.randint(-10, 40) if \"London\" in city else random.randint(0, 35)\n        humidity = random.randint(30, 90)\n        rainfall = random.uniform(0, 20)\n        \n        row = {\n            \"date\": (start_date + timedelta(days=i)).strftime(\"%Y-%m-%d\"),\n            \"city\": city,\n            \"temperature_c\": temp,\n            \"humidity_percent\": humidity,\n            \"rainfall_mm\": round(rainfall, 2)\n        }\n        data.append(row)\n\n# Write to CSV\nwith open(\"weather_data.csv\", \"w\", newline=\"\") as f:\n    writer = csv.DictWriter(f, fieldnames=[\"date\", \"city\", \"temperature_c\", \"humidity_percent\", \"rainfall_mm\"])\n    writer.writeheader()\n    writer.writerows(data)\n\nprint(f\"Generated {len(data)} weather records in weather_data.csv\")",
    "language": "python"
  }'

## 3. DATA ANALYSIS EXAMPLES

### Example 6: Analyze CSV Data (Sales Analysis)
curl -X POST http://localhost:8000/v1/execute \
  -H "Content-Type: application/json" \
  -d '{
    "code": "import pandas as pd\nimport matplotlib.pyplot as plt\n\n# Read and analyze sales data\ndf = pd.read_csv(\"sales_data.csv\")\n\n# Basic statistics\nprint(\"=== SALES DATA ANALYSIS ===\")\nprint(f\"Total records: {len(df)}\")\nprint(f\"Total sales: ${df[\"sales\"].sum():,.2f}\")\nprint(f\"Average sale: ${df[\"sales\"].mean():.2f}\")\nprint(f\"Top product: {df.groupby(\"product\")[\"sales\"].sum().idxmax()}\")\nprint(f\"Best region: {df.groupby(\"region\")[\"sales\"].sum().idxmax()}\")\n\n# Monthly sales trend\nif \"date\" in df.columns:\n    df[\"date\"] = pd.to_datetime(df[\"date\"])\n    df[\"month\"] = df[\"date\"].dt.strftime(\"%Y-%m\")\n    monthly_sales = df.groupby(\"month\")[\"sales\"].sum()\n    print(\"\\n=== MONTHLY SALES TREND ===\")\n    for month, sales in monthly_sales.items():\n        print(f\"{month}: ${sales:,.2f}\")\n\nprint(\"\\nAnalysis complete!\")",
    "language": "python"
  }'

### Example 7: Statistical Analysis with NumPy
curl -X POST http://localhost:8000/v1/execute \
  -H "Content-Type: application/json" \
  -d '{
    "code": "import numpy as np\nimport pandas as pd\n\n# Generate sample data\nnp.random.seed(42)\ndata = np.random.normal(100, 15, 1000)\n\nprint(\"=== STATISTICAL ANALYSIS ===\")\nprint(f\"Mean: {np.mean(data):.2f}\")\nprint(f\"Median: {np.median(data):.2f}\")\nprint(f\"Standard Deviation: {np.std(data):.2f}\")\nprint(f\"Min: {np.min(data):.2f}\")\nprint(f\"Max: {np.max(data):.2f}\")\nprint(f\"25th Percentile: {np.percentile(data, 25):.2f}\")\nprint(f\"75th Percentile: {np.percentile(data, 75):.2f}\")\n\n# Check for outliers\nq1, q3 = np.percentile(data, [25, 75])\niqr = q3 - q1\nlower_bound = q1 - 1.5 * iqr\nupper_bound = q3 + 1.5 * iqr\noutliers = data[(data < lower_bound) | (data > upper_bound)]\nprint(f\"Number of outliers: {len(outliers)}\")\n\nprint(\"\\nStatistical analysis complete!\")",
    "language": "python"
  }'

## 5. MACHINE LEARNING EXAMPLES

### Example 11: Simple Linear Regression
curl -X POST http://localhost:8000/v1/execute \
  -H "Content-Type: application/json" \
  -d '{
    "code": "import numpy as np\nimport matplotlib.pyplot as plt\nfrom sklearn.linear_model import LinearRegression\nfrom sklearn.metrics import r2_score\n\n# Generate sample data\nnp.random.seed(42)\nX = np.random.rand(100, 1) * 10\ny = 2.5 * X.flatten() + np.random.randn(100) * 2 + 5\n\n# Train linear regression model\nmodel = LinearRegression()\nmodel.fit(X, y)\n\n# Make predictions\ny_pred = model.predict(X)\n\n# Calculate metrics\nr2 = r2_score(y, y_pred)\n\nprint(\"=== LINEAR REGRESSION RESULTS ===\")\nprint(f\"Slope (coefficient): {model.coef_[0]:.3f}\")\nprint(f\"Intercept: {model.intercept_:.3f}\")\nprint(f\"R² Score: {r2:.3f}\")\n\n# Create visualization\nplt.figure(figsize=(10, 6))\nplt.scatter(X, y, alpha=0.6, label=\"Actual data\")\nplt.plot(X, y_pred, color=\"red\", linewidth=2, label=\"Regression line\")\nplt.title(\"Linear Regression: X vs Y\")\nplt.xlabel(\"X\")\nplt.ylabel(\"Y\")\nplt.legend()\nplt.grid(True, alpha=0.3)\nplt.tight_layout()\nplt.savefig(\"linear_regression.png\", dpi=300, bbox_inches=\"tight\")\nplt.close()\n\nprint(\"\\nVisualization saved as linear_regression.png\")",
    "language": "python"
  }'

### Example 12: K-Means Clustering
curl -X POST http://localhost:8000/v1/execute \
  -H "Content-Type: application/json" \
  -d '{
    "code": "import numpy as np\nimport matplotlib.pyplot as plt\nfrom sklearn.cluster import KMeans\nfrom sklearn.datasets import make_blobs\n\n# Generate sample data\nnp.random.seed(42)\nX, _ = make_blobs(n_samples=300, centers=4, cluster_std=0.60, random_state=0)\n\n# Perform K-means clustering\nkmeans = KMeans(n_clusters=4, random_state=42, n_init=10)\nclusters = kmeans.fit_predict(X)\n\n# Get cluster centers\ncenters = kmeans.cluster_centers_\n\nprint(\"=== K-MEANS CLUSTERING RESULTS ===\")\nprint(f\"Number of clusters: {len(np.unique(clusters))}\")\nprint(f\"Cluster centers:\")\nfor i, center in enumerate(centers):\n    print(f\"  Cluster {i}: ({center[0]:.2f}, {center[1]:.2f})\")\n\n# Create visualization\nplt.figure(figsize=(10, 6))\n\n# Plot data points colored by cluster\nfor i in range(4):\n    cluster_points = X[clusters == i]\n    plt.scatter(cluster_points[:, 0], cluster_points[:, 1], \n               label=f\"Cluster {i}\", alpha=0.6)\n\n# Plot cluster centers\nplt.scatter(centers[:, 0], centers[:, 1], \n           marker=\"x\", s=200, linewidths=3, \n           color=\"red\", label=\"Centroids\")\n\nplt.title(\"K-Means Clustering (k=4)\")\nplt.xlabel(\"Feature 1\")\nplt.ylabel(\"Feature 2\")\nplt.legend()\nplt.grid(True, alpha=0.3)\nplt.tight_layout()\nplt.savefig(\"kmeans_clustering.png\", dpi=300, bbox_inches=\"tight\")\nplt.close()\n\nprint(\"\\nVisualization saved as kmeans_clustering.png\")",
    "language": "python"
  }'
